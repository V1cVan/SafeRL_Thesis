273  template <typename COLLISION_fun>
274  bool
275  collision(
276  AABBtree const & tree,
277  COLLISION_fun ifun,
278  bool swap_tree = false
279  ) const {
280 
281  // check bbox with
282  if ( !tree.pBBox->collision(*pBBox) ) return false;
283 
284  int icase = (children.empty() ? 0 : 1) +
285  (tree.children.empty()? 0 : 2);
286 
287  switch ( icase ) {
288  case 0: // both leaf, use GeomPrimitive intersection algorithm
289  if ( swap_tree ) return ifun( tree.pBBox, pBBox );
290  else return ifun( pBBox, tree.pBBox );
291  case 1: // first is a tree, second is a leaf
292  { typename vector<PtrAABB>::const_iterator it;
293  for ( it = children.begin(); it != children.end(); ++it )
294  if ( tree.collision( **it, ifun, !swap_tree ) )
295  return true;
296  }
297  break;
298  case 2: // first leaf, second is a tree
299  { typename vector<PtrAABB>::const_iterator it;
300  for ( it = tree.children.begin();
301  it != tree.children.end(); ++it )
302  if ( this->collision( **it, ifun, swap_tree ) )
303  return true;
304  }
305  break;
306  case 3: // first is a tree, second is a tree
307  { typename vector<PtrAABB>::const_iterator c1;
308  typename vector<PtrAABB>::const_iterator c2;
309  for ( c1 = children.begin(); c1 != children.end(); ++c1 )
310  for ( c2 = tree.children.begin();
311  c2 != tree.children.end(); ++c2 )
312  if ( (*c1)->collision( **c2, ifun, swap_tree ) )
313  return true;
314  }
315  break;
316  }
317  return false;
318  }
